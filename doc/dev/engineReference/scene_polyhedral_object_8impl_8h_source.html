67 template< typename Base, typename P >
68 bool ScenePolyhedralObject< Base, P >::onAdd()
69 {
70  // If no polyhedron has been initialized for the zone, default
71  // to object box. Do this before calling the parent's onAdd()
72  // so that we set the object box correctly.
73 
74  if( mPolyhedron.getNumPlanes() == 0 )
75  {
76  mPolyhedron.buildBox( MatrixF::Identity, this->getObjBox() );
77  mIsBox = true;
78  }
79  else
80  {
81  mIsBox = false;
82 
83  // Compute object-space bounds from polyhedron.
84  this->mObjBox = mPolyhedron.getBounds();
85  }
86 
87  if( !Parent::onAdd() )
88  return false;
89 
90  return true;
91 }

95 template< typename Base, typename P >
96 bool ScenePolyhedralObject< Base, P >::containsPoint( const Point3F& point )
97 {
98  // If our shape is the OBB, use the default implementation
99  // inherited from SceneObject.
100 
101  if( this->mIsBox )
102  return Parent::containsPoint( point );
103 
104  // Take the point into our local object space.
105 
106  Point3F p = point;
107  this->getWorldTransform().mulP( p );
108  p.convolveInverse( this->getScale() );
109 
110  // See if the polyhedron contains the point.
111 
112  return mPolyhedron.isContained( p );
113 }

161 template< typename Base, typename P >
162 U32 ScenePolyhedralObject< Base, P >::packUpdate( NetConnection* connection, U32 mask, BitStream* stream )
163 {
164  U32 retMask = Parent::packUpdate( connection, mask, stream );
165 
166  if( stream->writeFlag( !mIsBox && ( mask & PolyMask ) ) )
167  {
168  // Write planes.
169 
170  const U32 numPlanes = mPolyhedron.getNumPlanes();
171  const typename PolyhedronType::PlaneType* planes = mPolyhedron.getPlanes();
172 
173  stream->writeInt( numPlanes, 8 );
174  for( U32 i = 0; i < numPlanes; ++ i )
175  mathWrite( *stream, planes[ i ] );
176 
177  // Write points.
178 
179  const U32 numPoints = mPolyhedron.getNumPoints();
180  const typename PolyhedronType::PointType* points = mPolyhedron.getPoints();
181 
182  stream->writeInt( numPoints, 8 );
183  for( U32 i = 0; i < numPoints; ++ i )
184  mathWrite( *stream, points[ i ] );
185 
186  // Write edges.
187 
188  const U32 numEdges = mPolyhedron.getNumEdges();
189  const typename PolyhedronType::EdgeType* edges = mPolyhedron.getEdges();
190 
191  stream->writeInt( numEdges, 8 );
192  for( U32 i = 0; i < numEdges; ++ i )
193  {
194  const typename PolyhedronType::EdgeType& edge = edges[ i ];
195 
196  stream->writeInt( edge.face[ 0 ], 8 );
197  stream->writeInt( edge.face[ 1 ], 8 );
198  stream->writeInt( edge.vertex[ 0 ], 8 );
199  stream->writeInt( edge.vertex[ 1 ], 8 );
200  }
201  }
202 
203  return retMask;
204 }

234  mPolyhedron.edgeList.setSize( numEdges );

267 {

293  }
