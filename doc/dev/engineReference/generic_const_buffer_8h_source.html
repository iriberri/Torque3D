288 // NOTE: These inlines below are here to get the very best possible
289 // performance when setting the device shader constants and can be
290 // called 4000-8000 times per frame or more.
291 //
292 // You need a very good reason to consider changing them.
293 
294 inline void GenericConstBuffer::internalSet( const GenericConstBufferLayout::ParamDesc &pd,
295  const GFXShaderConstType constType,
296  const U32 size,
297  const void *data )
298 {
299  // NOTE: We should have never gotten here if the buffer
300  // was null as no valid shader constant could have been
301  // assigned.
302  //
303  // If this happens its a bug in another part of the code.
304  //
305  AssertFatal( mBuffer, "GenericConstBuffer::internalSet - The buffer is NULL!" );
306 
307  if ( mLayout->set( pd, constType, size, data, mBuffer ) )
308  {
309  #ifdef TORQUE_DEBUG
310 
311  // Update the debug assignment tracking.
312  mWasAssigned[ pd.index ] = true;
313 
314  #endif
315 
316  // Keep track of the dirty range so it can be queried
317  // later in GenericConstBuffer::getDirtyBuffer.
318  mDirtyStart = getMin( pd.offset, mDirtyStart );
319  mDirtyEnd = getMax( pd.offset + pd.size, mDirtyEnd );
320  }
321 }

323 inline void GenericConstBuffer::setDirty( bool dirty )
324 {
325  if ( !mBuffer )
326  return;
327 
328  if ( dirty )
329  {
330  mDirtyStart = 0;
331  mDirtyEnd = mLayout->getBufferSize();
332  }
333  else if ( !dirty )
334  {
335  mDirtyStart = U32_MAX;
336  mDirtyEnd = 0;
337  }
338 }

340 inline const U8* GenericConstBuffer::getDirtyBuffer( U32 *start, U32 *size )
341 {
342  AssertFatal( isDirty(), "GenericConstBuffer::getDirtyBuffer() - Buffer is not dirty!" );
343  AssertFatal( mDirtyEnd > mDirtyStart, "GenericConstBuffer::getDirtyBuffer() - Dirty range is invalid!" );
344  AssertFatal( mBuffer, "GenericConstBuffer::getDirtyBuffer() - Buffer is empty!" );
345 
346  // Use the area we calculated during internalSet.
347  *size = mDirtyEnd - mDirtyStart;
348  *start = mDirtyStart;
349  const U8 *buffer = mBuffer + mDirtyStart;
350 
351  // Clear the dirty state while we're here.
352  mDirtyStart = U32_MAX;
353  mDirtyEnd = 0;
354 
355  return buffer;
356 }
