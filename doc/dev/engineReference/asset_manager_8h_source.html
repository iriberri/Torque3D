151  template<typename T> T* acquireAsset( const char* pAssetId )
152  {
153  // Sanity!
154  AssertFatal( pAssetId != NULL, "Cannot acquire NULL asset Id." );
155 
156  // Is this an empty asset Id?
157  if ( *pAssetId == 0 )
158  {
159  // Yes, so return nothing.
160  return NULL;
161  }
162 
163  // Find asset.
164  AssetDefinition* pAssetDefinition = findAsset( pAssetId );
165 
166  // Did we find the asset?
167  if ( pAssetDefinition == NULL )
168  {
169  // No, so warn.
170  Con::warnf( "Asset Manager: Failed to acquire asset Id '%s' as it does not exist.", pAssetId );
171  return NULL;
172  }
173 
174  // Is asset loading?
175  if ( pAssetDefinition->mAssetLoading == true )
176  {
177  // Yes, so we've got a circular loop which we cannot resolve!
178  Con::warnf( "Asset Manager: Failed to acquire asset Id '%s' as loading it involves a cyclic dependency on itself which cannot be resolved.", pAssetId );
179  return NULL;
180  }
181 
182  // Info.
183  if ( mEchoInfo )
184  {
185  Con::printSeparator();
186  Con::printf( "Asset Manager: Started acquiring Asset Id '%s'...", pAssetId );
187  }
188 
189  // Is the asset already loaded?
190  if ( pAssetDefinition->mpAssetBase == NULL )
191  {
192  // No, so info
193  if ( mEchoInfo )
194  {
195  // Fetch asset Id.
196  StringTableEntry assetId = StringTable->insert( pAssetId );
197 
198  // Find any asset dependencies.
199  typeAssetDependsOnHash::Iterator assetDependenciesItr = mAssetDependsOn.find( assetId );
200 
201  // Does the asset have any dependencies?
202  if ( assetDependenciesItr != mAssetDependsOn.end() )
203  {
204  // Yes, so show all dependency assets.
205  Con::printf( "Asset Manager: > Found dependencies:" );
206 
207  // Iterate all dependencies.
208  while( assetDependenciesItr != mAssetDependsOn.end() && assetDependenciesItr->key == assetId )
209  {
210  // Info.
211  Con::printf( "Asset Manager: > Asset Id '%s'", assetDependenciesItr->value );
212 
213  // Next dependency.
214  assetDependenciesItr++;
215  }
216  }
217  }
218 
219  // Flag asset as loading.
220  pAssetDefinition->mAssetLoading = true;
221 
222  // Generate primary asset.
223  pAssetDefinition->mpAssetBase = mTaml.read<T>( pAssetDefinition->mAssetBaseFilePath );
224 
225  // Flag asset as finished loading.
226  pAssetDefinition->mAssetLoading = false;
227 
228  // Did we generate the asset?
229  if ( pAssetDefinition->mpAssetBase == NULL )
230  {
231  // No, so warn.
232  Con::warnf( "Asset Manager: > Failed to acquire asset Id '%s' as loading the asset file failed to return the asset or the correct asset type: '%s'.",
233  pAssetId, pAssetDefinition->mAssetBaseFilePath );
234  return NULL;
235  }
236 
237  // Increase loaded count.
238  pAssetDefinition->mAssetLoadedCount++;
239 
240  // Info.
241  if ( mEchoInfo )
242  {
243  Con::printf( "Asset Manager: > Loading asset into memory as object Id '%d' from file '%s'.",
244  pAssetDefinition->mpAssetBase->getId(), pAssetDefinition->mAssetBaseFilePath );
245  }
246 
247  // Set ownership by asset manager.
248  pAssetDefinition->mpAssetBase->setOwned( this, pAssetDefinition );
249 
250  // Is the asset internal?
251  if ( pAssetDefinition->mAssetInternal )
252  {
253  // Yes, so increase internal loaded asset count.
254  if ( ++mLoadedInternalAssetsCount > mMaxLoadedInternalAssetsCount )
255  mMaxLoadedInternalAssetsCount = mLoadedInternalAssetsCount;
256  }
257  else
258  {
259  // No, so increase external loaded assets count.
260  if ( ++mLoadedExternalAssetsCount > mMaxLoadedExternalAssetsCount )
261  mMaxLoadedExternalAssetsCount = mLoadedExternalAssetsCount;
262  }
263  }
264  else if ( pAssetDefinition->mpAssetBase->getAcquiredReferenceCount() == 0 )
265  {
266  // Info.
267  if ( mEchoInfo )
268  {
269  Con::printf( "Asset Manager: > Acquiring from idle state." );
270  }
271  }
272 
273  // Set acquired asset.
274  T* pAcquiredAsset = dynamic_cast<T*>( (AssetBase*)pAssetDefinition->mpAssetBase );
275 
276  // Is asset the correct type?
277  if ( pAcquiredAsset == NULL )
278  {
279  // No, so warn.
280  Con::warnf( "Asset Manager: > Failed to acquire asset Id '%s' as it was not the required asset type: '%s'.", pAssetId, pAssetDefinition->mAssetBaseFilePath );
281  return NULL;
282  }
283 
284  // Acquire asset reference.
285  pAcquiredAsset->acquireAssetReference();
286 
287  // Info.
288  if ( mEchoInfo )
289  {
290  Con::printf( "Asset Manager: > Finished acquiring asset. Reference count now '%d'.", pAssetDefinition->mpAssetBase->getAcquiredReferenceCount() );
291  Con::printSeparator();
292  }
293 
294  return pAcquiredAsset;
295  }

298  template<typename T> T* acquireAsPrivateAsset( const char* pAssetId )
299  {
300  // Acquire the asset normally.
301  T* pAsset = acquireAsset<T>( pAssetId );
302 
303  // Finish if the asset was not acquired.
304  if ( pAsset == NULL )
305  return NULL;
306 
307  // Clone the asset.
308  T* pAssetClone = dynamic_cast<T*>( pAsset->clone() );
309 
310  // Sanity!
311  AssertFatal( pAssetClone != NULL, "acquireAsPrivateAsset() - Failed to clone asset type." );
312 
313  // Release the public asset.
314  releaseAsset( pAssetId );
315 
316  // Add as a private asset.
317  addPrivateAsset( pAssetClone );
318 
319  return pAssetClone;
320  }
