82  ZoneContentIterator( SceneZoneSpaceManager* manager, S32 zoneId, bool upToDate = true )
83  {
84  AssertFatal( zoneId < manager->getNumZones(), "SceneZoneSpaceManager::ZoneContentIterator - Zone ID out of range" );
85 
86  if( upToDate )
87  {
88  // Since zoning is updated lazily, the zone contents may actually
89  // be out of date. Force an update by triggering rezoning on the
90  // zone object. This is brute-force but this iterator is not meant
91  // to be used for high-frequency code anyway.
92  //
93  // Use the area-based rezoning so that we can also properly iterate
94  // over the contents of SceneRootZone.
95  manager->_rezoneObjects( ( ( SceneObject* ) manager->getZoneOwner( zoneId ) )->getWorldBox() );
96  }
97 
98  mCurrent = manager->mZoneLists[ zoneId ]->nextInBin; // Skip zone object entry.
99  }
100 
101  bool isValid() const
102  {
103  return ( mCurrent != NULL );
104  }
105  bool operator !() const
106  {
107  return ( mCurrent == NULL );
108  }
109  ZoneContentIterator& operator ++()
110  {
111  if( mCurrent )
112  mCurrent = mCurrent->nextInBin;
113  return *this;
114  }
115  ZoneContentIterator& operator --()
116  {
117  if( mCurrent )
118  mCurrent = mCurrent->prevInBin;
119  return *this;
120  }
121  SceneObject* operator *() const
122  {
123  AssertFatal( mCurrent != NULL, "SceneManager::ZoneContentIterator::operator* - Invalid iterator" );
124  return mCurrent->object;
125  }
126  SceneObject* operator ->() const
127  {
128  AssertFatal( mCurrent != NULL, "SceneManager::ZoneContentIterator::operator-> - Invalid iterator" );
